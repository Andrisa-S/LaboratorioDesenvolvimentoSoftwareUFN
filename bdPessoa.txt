CREATE DATABASE aula;

USE aula;

CREATE TABLE pessoa (
     id INT AUTO_INCREMENT PRIMARY KEY,
     nome VARCHAR(50),
     sexo VARCHAR(50),
     idioma VARCHAR(30)
     );

CREATE TABLE veiculo(
    id INT AUTO_INCREMENT PRIMARY KEY,
    modelo VARCHAR(30),
    placa VARCHAR(7),
    id_pessoa INT NOT NULL,
    foreign key(id_pessoa) references pessoa(id)
    );
